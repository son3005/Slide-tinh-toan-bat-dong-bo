# Mã giả: 
# def task(): ...

with Executor(max_workers=3) as executor: # Thread hoặc Process Pool
    futures = []
    for i in range(5):
        f = executor.submit(task, i)
        futures.append(f)
    
    print("Main thread!")
    
    for f in futures:
        print(f.result())
