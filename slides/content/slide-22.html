6. Ví dụ Code Demo: I/O-bound vs CPU-bound Task

from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor
import time

def tai_web(url):
    time.sleep(1)  # Giả lập trễ mạng I/O
    return f"Đã tải {url}"

def tinh_toan_nang(n):
    return sum(i * i for i in range(n))  # Nặng CPU

if __name__ == "__main__":
    urls = ["google.com", "facebook.com", "github.com"]
    inputs = [8M, 8M, 8M, 8M]
    
    # [I/O ThreadPool]
    with ThreadPoolExecutor() as ex:
        list(ex.map(tai_web, urls))
        
    # [CPU ProcessPool]
    with ProcessPoolExecutor() as ex:
        list(ex.map(tinh_toan_nang, inputs))
